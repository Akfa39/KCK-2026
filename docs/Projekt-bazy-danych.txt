Table TBL_Users {
  id integer [primary key, increment, not null]
  user varchar [not null, unique]
}

//Ustawienia i dane użytkownika
Table TBL_User_Settings {
  user_id integer [primary key]
  language varchar [not null, default: 'pl']
  unit_system varchar [not null, default: 'metric', note: 'metric | imperial']
  weight_kg real
  height_cm real
  birth_date date
  gender varchar [note: 'M | F']
}

}

Table TBL_Training_Plans {
  id integer [primary key, increment, not null]

}

//Plan trenigowy
Table TBL_Training_Plan_Exercises {
  id integer [primary key, increment]
  plan_id integer [not null]
  exercise_id integer [not null]
  order_index integer [not null]
  sets integer [not null, default: 1]
  target_reps integer
  target_duration_seconds integer
}

//Historia treningów

  id integer [primary key, increment]

  ended_at time
  calories_burned integer
  notes text
  status varchar [not null, default: 'completed', note: 'in_progress | completed | aborted']
}
